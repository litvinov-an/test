SELECT a.id 
FROM test as a
HAVING (SELECT sum(b.amount) 
    FROM test as b WHERE 
    b.date <= a.date
    ORDER BY b.date ASC) < 0 AND
        (SELECT sum(b.amount) 
    FROM test as b WHERE 
    b.date < a.date
    ORDER BY b.date ASC) > 0
ORDER BY a.date DESC
LIMIT 1
